\section{Introduction}

\begin{frame}{Background \& Motivation}
    \begin{columns}
        \begin{column}{0.6\textwidth}
            \textbf{Agentic Platforms:}
            \begin{itemize}
                \item Integrate autonomous/semi-autonomous agents into software development
                \item Leverage LLMs for code generation, debugging, refactoring, testing
                \item Growing industry adoption for enhanced developer productivity
            \end{itemize}

            \vspace{0.5cm}
            \textbf{The Challenge:}
            \begin{itemize}
                \item No systematic comparisons exist
                \item Practitioners lack evidence-based guidance
                \item Heterogeneous architectures make comparison difficult
            \end{itemize}
        \end{column}
        \begin{column}{0.35\textwidth}
            \begin{block}{Key Gap}
                How do practitioners select the right agentic platform for their needs?
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\section{Objectives}

\begin{frame}{Project Objectives}
    \textbf{Core Goals:}
    \begin{enumerate}
        \item Construct a systematic DESMET-based evaluation framework
        \item Evaluate 10+ platforms across multiple dimensions
        \item Identify comparative strengths and weaknesses
        \item Provide actionable guidance for practitioners
    \end{enumerate}

    \vspace{0.5cm}
    \textbf{Advanced Goals:}
    \begin{itemize}
        \item Develop a reusable evaluation framework for future platforms
        \item Create a taxonomy of agentic platform architectures
        \item Validate findings through stakeholder consultation
    \end{itemize}
\end{frame}

\section{Methodology}

\begin{frame}{DESMET Methodology}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \textbf{Hybrid Evaluation Approach:}
            \begin{itemize}
                \item \textbf{Benchmarking}: Standardised SE tasks across platforms
                \item \textbf{Qualitative Screening}: Feature-based assessment
            \end{itemize}

            \vspace{0.3cm}
            \textbf{Why Hybrid?}
            \begin{itemize}
                \item Architectural heterogeneity
                \item Task variability
                \item Developer experience matters
                \item Rapidly evolving landscape
            \end{itemize}
        \end{column}
        \begin{column}{0.45\textwidth}
            \begin{block}{DESMET Guidance}
                \small
                \begin{enumerate}
                    \item Feature identification
                    \item Test design (standardised)
                    \item Platform inclusion criteria
                    \item Hybrid evaluation framing
                \end{enumerate}
            \end{block}
        \end{column}
    \end{columns}
\end{frame}
